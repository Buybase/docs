---
title: "Create Summary"
api: "POST /summary/create"
description: "Create a summary of prices for a singular product."
---

## Request Headers

<ParamField header="authorization" type="string" required>
  Your Buybase API key
</ParamField>

## Request Body

<ParamField body="platform" type="string" required>
  The e-commerce platform. Currently supported: `amazon`
</ParamField>

<ParamField body="product_url" type="string" required>
  The URL of the product to checkout. Must be from a supported platform.
</ParamField>

<ParamField body="user_id" type="string" required>
  The Buybase user ID to create the checkout summary for.
</ParamField>

<ParamField body="app_id" type="string" required>
  Your Buybase app ID
</ParamField>

<ParamField body="developer_id" type="string" required>
  The unique identifier for the developer.
</ParamField>

## Response

<ResponseField name="message" type="string">
  Success message
</ResponseField>

<ResponseField name="code" type="string">
  Response code
</ResponseField>

<ResponseField name="summary" type="object">
  <Expandable title="properties">
    <ResponseField name="delivery_date" type="string">
      Estimated delivery date in ISO format
    </ResponseField>

    <ResponseField name="shipping_cost" type="number">
      Cost of shipping
    </ResponseField>

    <ResponseField name="buybase_fee_percent" type="number">
      Buybase fee percentage
    </ResponseField>

    <ResponseField name="agent_fee_percent" type="number">
      Agent fee percentage
    </ResponseField>

    <ResponseField name="unit_price" type="number">
      Price per unit of product
    </ResponseField>

    <ResponseField name="unit_fees_dollars" type="number">
      Total fees in dollars per unit.
      <Info>(`buybase_fee_percent` + `agent_fee_percent`) \* `unit_price`</Info>
    </ResponseField>

    <ResponseField name="unit_estimated_tax" type="number">
      Estimated tax per unit
    </ResponseField>

    <ResponseField name="unit_order_total" type="number">
      Total cost per unit including all fees and taxes
    </ResponseField>
  </Expandable>
</ResponseField>
